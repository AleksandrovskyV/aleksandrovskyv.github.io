//Variable

var thisComp = app.project.activeItem; // Переменная активной композиции
var compLayers = thisComp.layers;
var ActiveLayers = thisComp.selectedLayers; //Выделенный слой

//Process

for (var i = 1; i < compLayers.length+1; i++){ // Цикл выкл
thisComp.layer(i).shy = true;
}

for (var i = 0; i < ActiveLayers.length; i++){ // Цикл включения видимости выделеных слоев
ActiveLayers[i].shy = false;
}

//Final

thisComp.hideShyLayers = true; //Скрывает все невыделенные слои
